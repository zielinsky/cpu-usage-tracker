10static unsigned long cpu_usage(struct proc_stat prev,
11 struct proc_stat current) {
12 unsigned long prevIdle = prev.idle + prev.iowait;
13 unsigned long idle = current.idle + current.iowait;
14
15 unsigned long prevNonIdle = prev.user + prev.nice + prev.system + prev.irq +
16 prev.softirq + prev.steal;
17 unsigned long nonIdle = current.user + current.nice + current.system +
18 current.irq + current.softirq + current.steal;
19
20 unsigned long prevTotal = prevIdle + prevNonIdle;
21 unsigned long total = idle + nonIdle;
22
23 total -= prevTotal;
24 idle -= prevIdle;
25
